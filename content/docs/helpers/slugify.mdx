## Usage

```tsx
import { slugify } from "@/lib/slugify";

function App() {
  const title = "Hello World! This is a Test";
  const slug = slugify(title);

  return (
    <div>
      <p>Original: {title}</p>
      <p>Slugified: {slug}</p>
    </div>
  );
}
```

## API

### Parameters

| Parameter | Type     | Default | Description           |
| --------- | -------- | ------- | --------------------- |

### Returns

| Type     | Description         |
| -------- | ------------------- |
